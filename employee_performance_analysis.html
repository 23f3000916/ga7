# This script loads employee data, computes the frequency of the "IT" department,
# prints the count, creates a department distribution chart using matplotlib,
# and saves both the code and the visualization into a single self-contained HTML file.

import os
import io
import base64
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from datetime import datetime

# ---------- 1) Load the employee data ----------
# If a CSV named 'employee_performance.csv' exists in /mnt/data, we'll load it.
# Otherwise, we'll synthesize a 100-row dataset (including the 5 sample rows provided).

csv_path = "employee_performance.csv"  # relative path hint for local runs
if os.path.exists(csv_path):
    df = pd.read_csv(csv_path)
else:
    sample_rows = [
        {"employee_id":"EMP001","department":"Sales","region":"Latin America","performance_score":66.79,"years_experience":4,"satisfaction_rating":4.5},
        {"employee_id":"EMP002","department":"Finance","region":"Asia Pacific","performance_score":63.29,"years_experience":13,"satisfaction_rating":4.4},
        {"employee_id":"EMP003","department":"Operations","region":"North America","performance_score":68.52,"years_experience":9,"satisfaction_rating":3.6},
        {"employee_id":"EMP004","department":"R&D","region":"Asia Pacific","performance_score":84.57,"years_experience":1,"satisfaction_rating":3.4},
        {"employee_id":"EMP005","department":"Marketing","region":"Middle East","performance_score":87.36,"years_experience":12,"satisfaction_rating":4.7},
    ]
    df = pd.DataFrame(sample_rows)

    departments = ["Sales","Finance","Operations","R&D","Marketing","HR","IT","Legal","Support","Product"]
    regions = ["Asia Pacific","Europe","North America","Latin America","Middle East","Africa"]
    rng = np.random.default_rng(42)

    current_count = len(df)
    target_count = 100
    for i in range(current_count+1, target_count+1):
        dep = rng.choice(departments)
        reg = rng.choice(regions)
        perf = float(np.round(rng.uniform(50, 95), 2))
        yrs = int(rng.integers(0, 21))
        sat = float(np.round(rng.uniform(2.5, 5.0), 1))
        row = {
            "employee_id": f"EMP{i:03d}",
            "department": dep,
            "region": reg,
            "performance_score": perf,
            "years_experience": yrs,
            "satisfaction_rating": sat,
        }
        df.loc[len(df)] = row

# ---------- 2) Calculate the frequency count for the "IT" department ----------
it_count = int((df["department"] == "IT").sum())
print(f'Frequency count for "IT" department: {it_count}')

# ---------- 3) Create a histogram-like bar chart for department distribution ----------
dept_counts = df["department"].value_counts().sort_index()

plt.figure()
dept_counts.plot(kind="bar")
plt.xlabel("Department")
plt.ylabel("Number of Employees")
plt.title("Department Distribution")
plt.tight_layout()
plt.savefig("department_distribution.png", dpi=150, bbox_inches="tight")
plt.close()
